Enum — CategoriaProdutoEnum
Valores fixos para categorias
🤔 O que é um Enum?

Enum (enumeração) é um tipo especial que define um conjunto fixo de valores possíveis. Em vez de o usuário digitar qualquer texto no campo "categoria" (e escrever "eletronico", "eletrônico", "ELETRÔNICO"...), com enum ele só pode escolher os valores que definimos!

package br.com.sistemagestao.api.model;

import br.com.sistemagestao.api.enums.CategoriaProdutoEnum;
import jakarta.persistence.*;
import lombok.AllArgsConstructor;
import lombok.Data;
import lombok.NoArgsConstructor;

import java.math.BigDecimal;

@NoArgsConstructor
@AllArgsConstructor
@Data
@Table(name = "produto")
@Entity
public class Produto {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Column(name = "nome", nullable = false)
    private String nome;

    @Column(name = "descricao")
    private String descricao;

    @Column(name = "preco", nullable = false)
    private BigDecimal preco;

    @Column(name = "quantidade_estoque", nullable = false)
    private Integer quantidadeEstoque;

    @Enumerated(EnumType.STRING)  // Salva o NOME do enum (ex: "INFORMATICA")@Enumerated(EnumType.STRING)Salva o nome do enum como texto no banco. Sem isso, salvaria o número da posição (0, 1, 2...) — que é confuso e frágil.
    @Column(name = "categoria", nullable = false)
    private CategoriaProdutoEnum categoria;

    @ManyToOne  // MUITOS produtos para UM fornecedor@ManyToOneDefine o relacionamento: muitos Produtos pertencem a um Fornecedor. Cria a coluna fornecedor_id como chave estrangeira.
    @JoinColumn(name = "fornecedor_id", nullable = false)  // Nome da coluna FK@JoinColumn(name = "fornecedor_id")Define o nome da coluna de chave estrangeira no banco. O nullable=false obriga todo produto a ter um fornecedor.
    private Fornecedor fornecedor;
}
💡
Por que BigDecimal e não double para preço? O double tem problemas de precisão com decimais (famoso 0.1 + 0.2 = 0.30000000000000004). Para valores monetários, sempre use BigDecimal — ele calcula com precisão exata!

package br.com.sistemagestao.api.dto.produto;

import br.com.sistemagestao.api.enums.CategoriaProdutoEnum;
import jakarta.validation.constraints.*;

import java.math.BigDecimal;

public record ProdutoRequestDTO(

        @NotBlank(message = "O nome do produto é obrigatório")@NotBlankValida que o campo não é null, não é vazio ("") e não é só espaços ("   "). Perfeito para Strings obrigatórias!
        String nome,

        String descricao,  // sem validação — descrição é opcional!

        @NotNull(message = "O preço é obrigatório")@NotNullValida que o campo não é null. Usado para números e objetos. Para Strings, prefira @NotBlank.
        @Positive(message = "O preço deve ser maior que zero")@PositiveO valor deve ser maior que zero. Um produto não pode ter preço 0 ou negativo!
        BigDecimal preco,

        @NotNull(message = "A quantidade em estoque é obrigatória")@NotNullObrigatório informar a quantidade.
        @PositiveOrZero(message = "A quantidade em estoque não pode ser negativa")@PositiveOrZeroPermite zero (pode ter 0 em estoque) mas não permite negativo.
        Integer quantidadeEstoque,

        @NotNull(message = "A categoria é obrigatória")
        CategoriaProdutoEnum categoria,

        @NotNull(message = "O ID do fornecedor é obrigatório")
        Long fornecedorId   // só o ID! Não o objeto inteiro!
) {}
🎯
Por que fornecedorId e não fornecedor? O front não precisa enviar o objeto Fornecedor inteiro! Ele só manda o ID (um número). Quem busca o Fornecedor completo no banco é o ServiceImpl — essa é a responsabilidade do backend!

package br.com.sistemagestao.api.mapper;

import br.com.sistemagestao.api.dto.produto.ProdutoRequestDTO;
import br.com.sistemagestao.api.dto.produto.ProdutoResponseDTO;
import br.com.sistemagestao.api.model.Fornecedor;
import br.com.sistemagestao.api.model.Produto;

public class ProdutoMapper {

    // Converte RequestDTO → Entidade
    // Precisa do Fornecedor porque o DTO só tem o ID
    public static Produto toEntity(ProdutoRequestDTO dto, Fornecedor fornecedor) {

        Produto produto = new Produto();

        produto.setNome(dto.nome());          // record usa .nome() sem "get"!
        produto.setDescricao(dto.descricao());
        produto.setPreco(dto.preco());
        produto.setQuantidadeEstoque(dto.quantidadeEstoque());
        produto.setCategoria(dto.categoria());
        produto.setFornecedor(fornecedor);    // objeto completo, não o ID!

        return produto;
    }

    // Converte Entidade → ResponseDTO
    public static ProdutoResponseDTO toResponseDTO(Produto produto) {

        ProdutoResponseDTO dto = new ProdutoResponseDTO(
                produto.getId(),
                produto.getNome(),
                produto.getDescricao(),
                produto.getPreco(),
                produto.getQuantidadeEstoque(),
                produto.getCategoria(),
                produto.getFornecedor().getId(),          // pega o ID do fornecedor
                produto.getFornecedor().getNomeFantasia() // pega o nome do fornecedor
        );

        return dto;
    }
}
🔍
Repare na diferença: No toEntity, usamos dto.nome() (sem get, porque é record). No toResponseDTO, usamos produto.getNome() (com get, porque é classe normal com Lombok).

package br.com.sistemagestao.api.service.impl;

import br.com.sistemagestao.api.dto.produto.ProdutoRequestDTO;
import br.com.sistemagestao.api.dto.produto.ProdutoResponseDTO;
import br.com.sistemagestao.api.mapper.ProdutoMapper;
import br.com.sistemagestao.api.model.Fornecedor;
import br.com.sistemagestao.api.model.Produto;
import br.com.sistemagestao.api.repository.FornecedorRepository;
import br.com.sistemagestao.api.repository.ProdutoRepository;
import br.com.sistemagestao.api.service.ProdutoService;
import lombok.RequiredArgsConstructor;
import org.springframework.http.HttpStatus;
import org.springframework.stereotype.Service;
import org.springframework.web.server.ResponseStatusException;

import java.util.List;

@Service
@RequiredArgsConstructor  // Lombok gera o construtor com os campos "final"@RequiredArgsConstructorO Lombok gera um construtor com TODOS os campos marcados como final. O Spring usa esse construtor para injetar automaticamente. Substitui o @Autowired!
public class ProdutoServiceImpl implements ProdutoService {

    // Injeção de dependência pelo construtor (via Lombok)
    private final ProdutoRepository produtoRepository;
    private final FornecedorRepository fornecedorRepository;

    @Override
    public ProdutoResponseDTO criarProduto(ProdutoRequestDTO dto) {

        // Busca o fornecedor pelo ID. Se não existir, lança 404.
        Fornecedor fornecedor = fornecedorRepository.findById(dto.fornecedorId())
                .orElseThrow(() -> new ResponseStatusException(
                        HttpStatus.NOT_FOUND, "Fornecedor não encontrado"));

        // Converte DTO → Entidade usando o Mapper
        Produto produto = ProdutoMapper.toEntity(dto, fornecedor);
        produtoRepository.save(produto);

        // Converte Entidade → ResponseDTO e retorna
        return ProdutoMapper.toResponseDTO(produto);
    }

    @Override
    public List<ProdutoResponseDTO> listarTodosProdutos() {

        // Busca todos → stream → converte cada um → lista de DTOs
        return produtoRepository.findAll()
                .stream()
                .map(ProdutoMapper::toResponseDTO)
                .toList();
    }

    @Override
    public ProdutoResponseDTO buscarProdutoPorId(Long id) {

        Produto produto = produtoRepository.findById(id)
                .orElseThrow(() -> new ResponseStatusException(
                        HttpStatus.NOT_FOUND, "Produto não encontrado"));

        return ProdutoMapper.toResponseDTO(produto);
    }

    @Override
    public ProdutoResponseDTO atualizarProdutoPorId(Long id, ProdutoRequestDTO dto) {

        Produto produto = produtoRepository.findById(id)
                .orElseThrow(() -> new ResponseStatusException(
                        HttpStatus.NOT_FOUND, "Produto não encontrado"));

        Fornecedor fornecedor = fornecedorRepository.findById(dto.fornecedorId())
                .orElseThrow(() -> new ResponseStatusException(
                        HttpStatus.NOT_FOUND, "Fornecedor não encontrado"));

        produto.setNome(dto.nome());
        produto.setDescricao(dto.descricao());
        produto.setPreco(dto.preco());
        produto.setQuantidadeEstoque(dto.quantidadeEstoque());
        produto.setCategoria(dto.categoria());
        produto.setFornecedor(fornecedor);

        produtoRepository.save(produto);

        return ProdutoMapper.toResponseDTO(produto);
    }

    @Override
    public void deletarProdutoPorId(Long id) {

        Produto produto = produtoRepository.findById(id)
                .orElseThrow(() -> new ResponseStatusException(
                        HttpStatus.NOT_FOUND, "Produto não encontrado"));

        produtoRepository.deleteById(produto.getId());
    }
}

package br.com.sistemagestao.api.controller;

import br.com.sistemagestao.api.dto.produto.ProdutoRequestDTO;
import br.com.sistemagestao.api.dto.produto.ProdutoResponseDTO;
import br.com.sistemagestao.api.service.ProdutoService;
import jakarta.validation.Valid;
import lombok.RequiredArgsConstructor;
import org.springframework.http.HttpStatus;
import org.springframework.web.bind.annotation.*;

import java.util.List;

@RestController
@RequestMapping("/produtos")
@RequiredArgsConstructor
public class ProdutoController {

    private final ProdutoService produtoService;

    @PostMapping
    @ResponseStatus(HttpStatus.CREATED)
    public ProdutoResponseDTO criarProduto(@Valid @RequestBody ProdutoRequestDTO dto) {@ValidATIVA as validações do DTO! Sem o @Valid, as anotações @NotBlank, @Positive etc não fazem nada. É o "interruptor" das validações.
        return produtoService.criarProduto(dto);
    }

    @GetMapping
    @ResponseStatus(HttpStatus.OK)
    public List<ProdutoResponseDTO> listarTodosProdutos() {
        return produtoService.listarTodosProdutos();
    }

    @GetMapping("/{id}")
    @ResponseStatus(HttpStatus.OK)
    public ProdutoResponseDTO buscarProdutoPorId(@PathVariable Long id) {
        return produtoService.buscarProdutoPorId(id);
    }

    @PutMapping("/{id}")
    @ResponseStatus(HttpStatus.OK)
    public ProdutoResponseDTO atualizarProdutoPorId(@PathVariable Long id,
                                                     @Valid @RequestBody ProdutoRequestDTO dto) {
        return produtoService.atualizarProdutoPorId(id, dto);
    }

    @DeleteMapping("/{id}")
    @ResponseStatus(HttpStatus.NO_CONTENT)
    public void deletarProdutoPorId(@PathVariable Long id) {
        produtoService.deletarProdutoPorId(id);
    }
}
